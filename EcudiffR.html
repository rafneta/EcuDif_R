Ecuaciones diferenciales con R

Intalamos el paquete

install.packages('deSolve')

Resolvemos la siguiente ecuacion diferencial

\[ \begin{eqnarray} \dot{x}=-\dfrac{8}{3}x+yz\\ \dot{y}=-10(y-z)\\ \dot{z}=-xy+28y-z \end{eqnarray} \]

parametros <- c(a = -8/3,
                b = -10,
                c = 28)

estado <- c(X = 1,
            Y = 1,
            Z = 1)

Lorenz <- function (t, estado, parametros){
  with(as.list(c(estado, parametros)),{
# Razon de cambio
    dX <- a * X + Y * Z
    dY <- b * (Y-Z)
    dZ <- -X * Y + c * Y - Z
    list(c(dX, dY, dZ))
}) # temina with
} # termina funcion

tiempo <- seq(0, 100, by = 0.01)# tiempo de simulacion

Resolvemos con el paquete deSolve

library(deSolve)

solucion <- ode(y = estado,
                times = tiempo,
                func = Lorenz,
                parms = parametros)
